## Summary

Add a new `<ChartGroupedBar>` UI primitive in
`web/src/components/ui/chart-grouped-bar.tsx` -- pure-SVG
grouped bar chart. Each category gets its own horizontal
slot; within the slot, every visible series renders its
bar side-by-side. Distinct from `<ChartBar>` (11.438) in
that this primitive emphasises multi-series side-by-side
grouping + a clickable legend that toggles per-series
visibility (controlled or uncontrolled). Hover surfaces
the bar's category + series label + formatted value.
Optional `showValues` paints the formatted value above
each bar.

## API

```ts
interface ChartGroupedBarSeries {
  id: string;
  label: string;
  data: readonly number[];   // length should match categories.length
  color?: string;
}

interface ChartGroupedBarProps {
  categories: readonly string[];
  series: readonly ChartGroupedBarSeries[];
  width?: number;                      // default 560
  height?: number;                     // default 280
  padding?: number;                    // default 36
  groupGap?: number;                   // default 16
  barGap?: number;                     // default 2
  hiddenSeries?: readonly string[];    // controlled
  defaultHiddenSeries?: readonly string[];
  onHiddenSeriesChange?: (hidden: string[]) => void;
  showLegend?: boolean;                // default true
  showAxisTicks?: boolean;             // default true
  showGrid?: boolean;                  // default true
  showLabels?: boolean;                // default true (x labels)
  showValues?: boolean;                // default false (above bars)
  showTooltip?: boolean;               // default true
  animate?: boolean;                   // default true
  className?: string;
  ariaLabel?: string;                  // default 'Grouped bar chart'
  ariaDescription?: string;
  formatValue?: (v: number) => string;
  tickCount?: number;                  // default 5
  legendPlacement?: 'right' | 'bottom'; // default 'bottom'
  onBarClick?: (args: {
    series: ChartGroupedBarSeries;
    seriesIndex: number;
    category: string;
    categoryIndex: number;
    value: number;
  }) => void;
  onSeriesClick?: (args: {
    series: ChartGroupedBarSeries;
    index: number;
    hidden: boolean;
  }) => void;
}
```

## Behaviour

- Renders a `<div role="region">` wrapping `<svg
  role="img">` carrying the same `aria-label`.
- Layout: `computeGroupedBarLayout` walks every
  category, computes the slot width =
  `(innerWidth - groupGap * (n - 1)) / n`, and
  distributes the visible series side-by-side within
  the slot at `slotWidth - barGap * (m - 1) / m`.
  Hidden series collapse the slot evenly (more
  width per visible bar).
- Y scale: `getGroupedBarMaxValue` returns the
  largest positive finite value across visible
  series; falls back to 1 for empty / all-non-finite
  data.
- Non-finite / non-positive values clamp to 0 so a
  bar's height never goes negative.
- Hidden series: controlled via `hiddenSeries` +
  `onHiddenSeriesChange`, or uncontrolled via
  `defaultHiddenSeries`. Every legend entry is a
  focusable `<button aria-pressed>` -- clicking
  toggles the series's hidden state and fires
  `onSeriesClick`.
- Tooltip: hover opens a tooltip with the category,
  series label, and formatted value.
- Value labels: when `showValues=true`, a small
  caption above each non-zero bar shows the
  formatted value.
- Mount animation:
  `motion-safe:animate-fade-in` per bar group.
  `animate=false` removes the class.
- Default ARIA description: `"Grouped bar chart
  with N categories, M visible series. <label> sum
  <value> (n=<count>), ..."` -- empty input
  returns `"No data"`; `ariaDescription` overrides.

## Pure helpers (exported)

```ts
DEFAULT_CHART_GROUPED_BAR_WIDTH       = 560
DEFAULT_CHART_GROUPED_BAR_HEIGHT      = 280
DEFAULT_CHART_GROUPED_BAR_PADDING     = 36
DEFAULT_CHART_GROUPED_BAR_GROUP_GAP   = 16
DEFAULT_CHART_GROUPED_BAR_BAR_GAP     = 2
DEFAULT_CHART_GROUPED_BAR_TICK_COUNT  = 5

getGroupedBarMaxValue(series, hidden): number
computeGroupedBarLayout(categories, series, hidden,
                        innerW, innerH, padX, padY,
                        groupGap, barGap): GroupedBarRect[]
getGroupedBarTicks(max, count?): number[]
describeGroupedBarChart(categories, series, hidden,
                        formatValue?): string
```

- `getGroupedBarMaxValue` honours the `hidden` set;
  empty / all-non-finite -> 1; non-finite / non-
  positive values are skipped.
- `computeGroupedBarLayout` is the canonical layout
  pass. Empty inputs return `[]`. Hidden series do
  not consume slot space (the remaining bars expand
  to fill the slot).
- `getGroupedBarTicks` is the standard chart-tick
  helper (evenly-spaced; `[0]` for non-positive max;
  min count 2).
- `describeGroupedBarChart` produces the default
  ARIA summary including the visible-series count
  and per-series running totals; honours
  `formatValue`.
